Plot the Reference Model

true_mod;
truemodrs = reshape(true_mod,2,2)';
imagesc(truemodrs); colorbar; title('Reference Model (m/s)')

Solve for the best fitting Slowness/Velocity Model

tomo_mod_slow = inv(G'*G)*G'*d;
tomo_mol_vel = 1./tomo_mod_slow;
tomomodrs = reshape(tomo_mod_vel,2,2)';
imagesc(tomomodrs); colorbar; title('Best fitting model (m/s)')

Plot Best-fitting model as a % Difference from the Reference Model

pd = tomo_mod_vel./true_mod;
f=find(pd>=1);
pd(f)=(pd(f)-1)*100;
f=find(pd<=1);
pd(f)=-(1-pd(f))*100;
pd;
velperturbation = reshape(pd,2,2)';
imagesc(velperturbation); colorbar; title('Percent Difference from Reference Model')
